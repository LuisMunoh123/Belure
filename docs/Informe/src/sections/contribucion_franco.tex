\begin{taskdone}{Migración del sistema de menús a \texttt{getopt()}}
    Reemplazo del sistema de menús interactivo por un sistema basado en la función \texttt{getopt()} de C, permitiendo el uso de argumentos de línea de comandos.
\end{taskdone}

\begin{taskdone}{Implementación de algoritmos de búsqueda}
    Implementación de los algoritmos de búsqueda.
    \begin{itemize}
        \item Creación de rama \texttt{search\_algorithms}
        \item Búsqueda binaria recursiva estándar
        \item Búsqueda binaria para rangos
        \item Búsqueda exponencial
        \item Búsqueda por interpolación
        \item Merge a rama principal
    \end{itemize}
\end{taskdone}
